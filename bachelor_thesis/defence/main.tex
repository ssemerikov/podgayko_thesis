% ----- 12. Аудит SHA-256 -----
\begin{frame}{Аудит з SHA-256 хеш-ланцюгом}
\textbf{Кожен запис:} \texttt{\{actor, action, resource, before, after, prev\_hash, hash\}}\\[0.5em]
\textbf{Хеш:} $h_n = \mathrm{SHA{-}256}(h_{n-1} \,\|\, \mathrm{canonical\_json}(\text{запис}_n))$\\[0.3em]
\begin{itemize}\small
\item Append-only тригер блокує \texttt{UPDATE}/\texttt{DELETE}
\item \texttt{audit.verify()} перевіряє цілісність ланцюга на~10\,000 синтетичних upsert
\item Канонічна JSON-проєкція стабільна між~PostgreSQL JSONB round-trip
\item Криптографічний доказ для~зовнішнього перевіряючого
\end{itemize}
\textbf{Імплікації:} вищий рівень доказовості, ніж~у~всіх~5~національних системах; відповідає Закону «Про~освіту» (моніторинг якості) та~Закону про~академічну доброчесність.
\end{frame}

% ----- 13. Евалюація -----
\begin{frame}{Емпірична евалюація}
\begin{itemize}\small
\item \textbf{Аудит-ланцюг:} 10\,000 upsert + \texttt{verify()}~$\to$~100\% цілісність
\item \textbf{ABAC:} матриця role~$\times$~action~$\times$~resource ($\sim$160~тестів) + e2e~no-bypass
\item \textbf{Excel round-trip:} 8~шаблонів, fidelity 100\% для~валідних рядків
\item \textbf{Латентність:} цільовий $p_{95}<500$~мс (50~закладів~$\times$~1\,000~здобувачів)
\item \textbf{WCAG~2.1~AA:} axe-core на~9~маршрутах, 0~критичних порушень
\item \textbf{Експертне оцінювання $n\geq 3$:} SUS + 8 domain-питань + 4 open-feedback + 1--2 half-day сесії
\item \textbf{Walkthrough:} 5~сценаріїв, включно з~кейсом переміщеного закладу
\end{itemize}
\end{frame}

% ----- 14. Дорожня карта -----
\begin{frame}{Дорожня карта впровадження}
\begin{itemize}\small
\item \textbf{6~місяців} (\textit{воєнний час}): розгортання у~НМЦ ПТО Луганщини; імпорт із~Аналітичного збірника 2025; завершення експертного оцінювання
\item \textbf{12~місяців} (\textit{повоєнне відновлення}): пілот у~3--5~областях; інтеграція з~ЄДЕБО (ETL); КЕП-підпис; розширення PPP за~моделями ЄС
\item \textbf{24~місяці} (\textit{гармонізація з~ЄС}): сумісність з~10~індикаторами EQAVET; ML-передбачення відрахування з~explainability; поширення на~всі~25~областей; передача МОН
\end{itemize}
\textbf{Адресні рекомендації:} МОН (стандарт переміщених, EQAVET); обласні департаменти (on-prem, risk-based); керівники (юр./факт.\ адреса, Excel-bulk); інспектори (ABAC спеціалізація, обмін досвідом).
\end{frame}
